lang 说明

lang

_.castArray(value)

说明 如果 value 不是数组, 那么强制转为数组。

版本 4.4.0

参数

  1. value (*): 要处理的值。

返回

(Array): 返回转换后的数组。

  • 示例
  • 源码 
                        
_.castArray(1);
// => [1]
    
_.castArray({ 'a': 1 });
// => [{ 'a': 1 }]
    
_.castArray('abc');
// => ['abc']
    
_.castArray(null);
// => [null]
    
_.castArray(undefined);
// => [undefined]
    
_.castArray();
// => []
    
var array = [1, 2, 3];
console.log(_.castArray(array) === array);
// => true
                        
                    
                        
/**
* Casts `value` as an array if it's not one.
*
* @static
* @memberOf _
* @since 4.4.0
* @category Lang
* @param {*} value The value to inspect.
* @returns {Array} Returns the cast array.
* @example
*
* _.castArray(1);
* // => [1]
*
* _.castArray({ 'a': 1 });
* // => [{ 'a': 1 }]
*
* _.castArray('abc');
* // => ['abc']
*
* _.castArray(null);
* // => [null]
*
* _.castArray(undefined);
* // => [undefined]
*
* _.castArray();
* // => []
*
* var array = [1, 2, 3];
* console.log(_.castArray(array) === array);
* // => true
*/
function castArray() {
    if (!arguments.length) {
        return [];
    }
    var value = arguments[0];
    return isArray(value) ? value : [value];
}
