    line-height: $line-height-base;
    background-color: $thumbnail-bg;

    @include img-responsive;
    margin-left: auto;
    margin-right: auto;
    }

    // Image captions
    .caption {
    padding: $thumbnail-caption-padding;
    color: $thumbnail-caption-color;
    }
}
